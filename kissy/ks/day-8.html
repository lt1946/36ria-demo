在上一篇的教程中，明河曾说到给拆合菜单加动画，现在来看完整代码： 

(function(S){
	S.use('core',function(){
		var Event = S.Event,DOM = S.DOM,h;		
        //kissy的height\width方法，当元素的display:none时，获取的高度\高度为负数，期待新的版本可以修正这个问题				
		function height($el){
			var h = 0,display = $el.css('display'),position = $el.css('position'),visibility = $el.css('hidden');
			if($el.css('display') == 'none'){
				//TODO:kissy的css()存在bug，不返回KISSY.Node，导致断链
				$el.css({'display':'block','position':'absolute','visibility':'hidden'});
				h = $el.height();
				$el.css({'display':'none','position':position,'visibility':visibility});
			}else{
				h = $el.height();
			}
			return h;	
		}
		//获取菜单元素
		var $menu = S.one('#menu');
		//TODO:给变量加$，是明河写jquery时保留下的习惯，用加$的变量表示待操作的元素，在kissy中只有方法返回的是Node或NodeList实例（比如one、all）时，才加$前缀
		
		//子菜单
		var $childrenMenus = S.all('#menu ul');
		//隐藏所有子菜单
		$childrenMenus.hide();
		var $menuItemHeader = S.all("#menu div");
		

		$menuItemHeader.on('click',function(ev){
			//给子菜单标题，删除箭头样式
			S.all('#menu .arrow-down').removeClass('arrow-down');			
			var $menu = S.one(ev.target).next(),$arrow = S.one(ev.target).children(),
			    display = $menu.css('display');	

			//根据子菜单的显示情况来判断是否显示子菜单，同时改变箭头方向
			if(display == 'none'){
				h = height($menu);
				$menu.css({height: 0,display: 'block',overflow:'hidden'});
				//TODO:不加缓动和加缓动，速度差异明显
				$menu.animate({'height':h+'px'},1,'elasticOut');
				$arrow.addClass('arrow-down');
			}else{
				h = $menu.height();
				$menu.animate('height:0',0.3,S.Easing.easeNone,function(){
					$menu.hide().height(h);
				});							
				$arrow.removeClass('arrow-down');
			}			
			return false;		
		})
																																																		
	});	
})(KISSY);

kissy的动画类存在一些问题！调用起来不是非常顺手，这是个不少的问题。

为什么要加这个函数？

因为kissy的height\width方法，当元素的display:none时，获取的高度\高度为负数，有兴趣的朋友可以测试下。而拆合菜单的子菜单默认是不显示了，没办法明河只好临时写了这个函数应急下。

anim-node-plugin模块存在bug！

当调用show、hide、slideDown、slideUp、toggle时有时会报错，更关键上写法上有缺陷，当元素没有设置高度时，作者为了动画处理方便，使用height:auto/width:auto！而不是去获取元素的真实高度，当height、width被设置成auto，不可避免存在一些风险，有兴趣的朋友可以把代码中的动画部分改成slideDown、slideUp或toggle试下。这方面的问题，明河日后会专门写篇文章介绍。

尽管anim-node-plugin存在问题，明河还是介绍下几个常用方法。

1、 animate方法

这是颇为常用的方法，对应KISSY. Anim，所以 KISSY. Anim的几个参数 animate方法都有，比如 动画属性、 动画时长 、缓动函数、回调函数。

写法举例（可以看拆合菜单部分代码） 

h = $menu.height();
$menu.animate('height:0',0.3,S.Easing.easeNone,function(){
    $menu.hide().height(h);
});

2、fadeIn/fadeOut方法

常见的淡入淡出效果

Event.on('#demo-2','click',function(evt){
    var $div = S.one("#demo-2-div");
    $div.css({'opacity':'0'})
    //使用hide的方法直接报错
    if($div.css("display") == 'none'){
        $div.fadeIn(0.5,function(){
            $div.css({'display':'block'});
        });
    }else{
        $div.fadeOut(0.5,function(){
            $div.css({'display':'none'});
        });
    }
})

多点击几下，你就会发现，容器的宽度和高度莫名其妙丢了，原来是函数将height\width设置成了auto..........
